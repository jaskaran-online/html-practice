üíª Code Examples

/* Multi-line flex container */
.container {
    display: flex;
    flex-wrap: wrap;
    align-content: center;
    height: 400px;
}

/* Image gallery with centered rows */
.gallery {
    display: flex;
    flex-wrap: wrap;
    align-content: center;
    justify-content: space-around;
    min-height: 300px;
}

/* Tag cloud with space between rows */
.tags {
    display: flex;
    flex-wrap: wrap;
    align-content: space-between;
    height: 200px;
}

/* Card grid with stretched rows */
.cards {
    display: flex;
    flex-wrap: wrap;
    align-content: stretch;
    min-height: 500px;
}
