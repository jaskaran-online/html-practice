üíª Code Examples

/* Allow items to wrap */
.container {
    display: flex;
    flex-wrap: wrap;
}

/* Shorthand: flex-flow combines direction and wrap */
.container {
    display: flex;
    flex-flow: row wrap;
    /* Same as: flex-direction: row; flex-wrap: wrap; */
}

/* Responsive card layout */
.card-container {
    display: flex;
    flex-wrap: wrap;
    gap: 1rem;
}

.card {
    flex: 1 1 300px; /* grow, shrink, basis */
    min-width: 250px;
}
